---
title: "List Properties"
description: "Retrieve a paginated list of properties with filtering, sorting, and search"
---

## Endpoint

```
GET /api/properties
```

## Authentication

This endpoint requires authentication. All authenticated users (Viewer, Editor, Admin) can access this endpoint.

**Headers:**
- `Authorization: Bearer <token>` - JWT token from Supabase authentication

## Query Parameters

### Pagination

<ParamField query="page" type="number" default="1">
  Page number for pagination (minimum: 1)
</ParamField>

<ParamField query="limit" type="number" default="20">
  Number of results per page (minimum: 1, maximum: 100)
</ParamField>

### Sorting

<ParamField query="sortBy" type="string" default="createdAt">
  Field to sort by. Options:
  - `createdAt` - Sort by creation date
  - `updatedAt` - Sort by last update
  - `price` - Sort by property price
  - `qualityScore` - Sort by quality score
  - `bedrooms` - Sort by number of bedrooms
  - `daysOnMarket` - Sort by days on market
</ParamField>

<ParamField query="sortOrder" type="string" default="desc">
  Sort order: `asc` or `desc`
</ParamField>

### Filters

<ParamField query="listingType" type="string | array">
  Filter by listing type(s). Can be a single value or array. Options:
  - `SALE`
  - `RENT`
  - `LEASE`
  - `SHORTLET`
</ParamField>

<ParamField query="category" type="string | array">
  Filter by property category. Can be a single value or array. Options:
  - `RESIDENTIAL`
  - `COMMERCIAL`
  - `LAND`
  - `SHORTLET`
  - `INDUSTRIAL`
</ParamField>

<ParamField query="status" type="string">
  Filter by property status. Options:
  - `AVAILABLE`
  - `SOLD`
  - `RENTED`
  - `UNDER_OFFER`
  - `WITHDRAWN`
  - `EXPIRED`
</ParamField>

<ParamField query="verificationStatus" type="string">
  Filter by verification status. Options:
  - `UNVERIFIED`
  - `VERIFIED`
  - `FLAGGED`
  - `REJECTED`
</ParamField>

<ParamField query="minPrice" type="number">
  Minimum price filter (minimum: 0)
</ParamField>

<ParamField query="maxPrice" type="number">
  Maximum price filter (minimum: 0)
</ParamField>

<ParamField query="minBedrooms" type="number">
  Minimum number of bedrooms
</ParamField>

<ParamField query="maxBedrooms" type="number">
  Maximum number of bedrooms
</ParamField>

<ParamField query="minBathrooms" type="number">
  Minimum number of bathrooms
</ParamField>

<ParamField query="state" type="string">
  Filter by Nigerian state (e.g., "Lagos")
</ParamField>

<ParamField query="area" type="string | array">
  Filter by area(s). Can be a single value or array. Case-insensitive search.
</ParamField>

<ParamField query="lga" type="string">
  Filter by Local Government Area. Case-insensitive search.
</ParamField>

<ParamField query="siteId" type="string">
  Filter by source site ID
</ParamField>

<ParamField query="propertyType" type="string">
  Filter by property type (e.g., "Detached Duplex", "Flat / Apartment"). Case-insensitive search.
</ParamField>

<ParamField query="furnishing" type="string">
  Filter by furnishing status. Options:
  - `FURNISHED`
  - `SEMI_FURNISHED`
  - `UNFURNISHED`
  - `UNKNOWN`
</ParamField>

<ParamField query="isPremium" type="boolean">
  Filter premium properties only
</ParamField>

<ParamField query="isFeatured" type="boolean">
  Filter featured properties only
</ParamField>

<ParamField query="minQualityScore" type="number">
  Minimum quality score (0-100)
</ParamField>

<ParamField query="search" type="string">
  Full-text search across title, description, location, area, and agent name
</ParamField>

```javascript TypeScript
interface ListPropertiesParams {
  page?: number;
  limit?: number;
  sortBy?: 'createdAt' | 'updatedAt' | 'price' | 'qualityScore' | 'bedrooms' | 'daysOnMarket';
  sortOrder?: 'asc' | 'desc';
  listingType?: string | string[];
  category?: string | string[];
  status?: string;
  verificationStatus?: string;
  minPrice?: number;
  maxPrice?: number;
  minBedrooms?: number;
  maxBedrooms?: number;
  minBathrooms?: number;
  state?: string;
  area?: string | string[];
  lga?: string;
  siteId?: string;
  propertyType?: string;
  furnishing?: string;
  search?: string;
  isPremium?: boolean;
  isFeatured?: boolean;
  minQualityScore?: number;
}

const params: ListPropertiesParams = {
  page: 1,
  limit: 20,
  listingType: 'SALE',
  category: 'RESIDENTIAL',
  minBedrooms: 3,
  state: 'Lagos',
  area: 'Lekki',
  sortBy: 'price',
  sortOrder: 'asc'
};

const response = await fetch(
  `https://realtors-practice-new-api.onrender.com/api/properties?${new URLSearchParams(params as any)}`,
  {
    headers: {
      'Authorization': `Bearer ${token}`
    }
  }
);

const data = await response.json();
```

## Use Cases

### Search for properties in a specific area
```javascript
const properties = await fetch(
  'https://realtors-practice-new-api.onrender.com/api/properties?' + new URLSearchParams({
    area: 'Lekki',
    state: 'Lagos',
    listingType: 'SALE'
  }),
  { headers: { Authorization: `Bearer ${token}` } }
);
```

### Filter by price range
```javascript
const properties = await fetch(
  'https://realtors-practice-new-api.onrender.com/api/properties?' + new URLSearchParams({
    minPrice: '50000000',
    maxPrice: '100000000',
    listingType: 'SALE'
  }),
  { headers: { Authorization: `Bearer ${token}` } }
);
```

### Full-text search
```javascript
const properties = await fetch(
  'https://realtors-practice-new-api.onrender.com/api/properties?' + new URLSearchParams({
    search: 'swimming pool lekki'
  }),
  { headers: { Authorization: `Bearer ${token}` } }
);
```

### Filter premium properties
```javascript
const properties = await fetch(
```
